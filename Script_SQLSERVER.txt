/* SCRIPT PARA A CRIAÇÃO DA TABELA NO SQL SERVER */

/* CRIAÇÃO DO BANCO DE DADOS */
CREATE DATABASE CADCARDB
  
/* SELEÇÃO DO BANCO DE DADOS */
USE CADCARDB
  
/* CRIAÇÃO DA ENTIDADE VEICULO */
CREATE TABLE VEICULO
(
      ID INT NOT NULL IDENTITY,
      MARCA CHAR(40) NOT NULL,
      MODELO CHAR(50) NOT NULL,
      COR CHAR(30) NOT NULL,
      ANO INT NOT NULL,
      PRECO DECIMAL(15,2) NOT NULL,
      DESCR VARCHAR(500) NULL,
      NOVO BIT NOT NULL,
	  DTREGISTRO DATETIME2 NOT NULL,
	  DTATUALIZACAO DATETIME2 NULL
)
